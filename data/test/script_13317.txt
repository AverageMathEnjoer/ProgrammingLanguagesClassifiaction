b'/*\n * @lc app=leetcode id=3 lang=cpp\n *\n * [3] Longest Substring Without Repeating Characters\n */\n\n// @lc code=start\nclass Solution {\npublic:\n    int lengthOfLongestSubstring(string s) {\n        int n = s.size();\n        if(s.size()==0)return 0;\n        int i,j;\n        i=0, j=0;\n        vector<int> cnt(326, 0);\n        cnt[s[0]]++;\n        int ans=1;\n        while(1){\n            if(j==n-1) break;\n            if(cnt[s[j+1]] == 0) j++, cnt[s[j]]++, ans=max(ans,j-i+1);\n            else {\n                cnt[s[i++]]--;\n            }\n        }\n        return ans;\n    }\n};\n// @lc code=end\n\n'
