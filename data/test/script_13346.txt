b'-- | This module can be used by providers to perform colorful/formatted\n-- output and possibly re-use tasty\'s own output formats.\n--\n-- @since 1.3.1\nmodule Test.Tasty.Providers.ConsoleFormat\n  ( ResultDetailsPrinter(..)\n  , ConsoleFormat(..)\n  , ConsoleFormatPrinter\n  , noResultDetails\n  , failFormat\n  , infoFailFormat\n  , infoOkFormat\n  , okFormat\n  , skippedFormat\n  )\nwhere\n\nimport System.Console.ANSI\n\n-- | Console output format\n--\n-- @since 1.3.1\ndata ConsoleFormat = ConsoleFormat\n  { consoleIntensity :: ConsoleIntensity\n  , colorIntensity   :: ColorIntensity\n  , color            :: Color\n  }\n\n-- | Type of console format printer functions\n--\n-- @since 1.3.1\ntype ConsoleFormatPrinter\n  =  ConsoleFormat -- ^ selected console format\n  -> IO ()         -- ^ action to be executed with active console format\n  -> IO ()\n\n-- | Noop result details printer. The default for most providers.\n--\n-- @since 1.3.1\nnoResultDetails :: ResultDetailsPrinter\nnoResultDetails = ResultDetailsPrinter . const . const $ return ()\n\n-- | An action that prints additional information about a test using\n-- colors/formatting; see \'Test.Tasty.Providers.testFailedDetails\' and\n-- \'Test.Tasty.Runners.resultDetailsPrinter\'.\n--\n-- As input, this action is provided with the current indentation level and\n-- a \'ConsoleFormatPrinter\', which tells it how perform output.\n--\n-- This is a newtype to allow a \'Show\' instance.\n--\n-- @since 1.3.1\nnewtype ResultDetailsPrinter = ResultDetailsPrinter\n  (Int -> ConsoleFormatPrinter -> IO ())\n\ninstance Show ResultDetailsPrinter where\n  show _printer = "ResultDetailsPrinter"\n\n-- | Format used to display failures\n--\n-- @since 1.3.1\nfailFormat :: ConsoleFormat\nfailFormat = ConsoleFormat BoldIntensity   Vivid Red\n\n-- | Format used to display additional information on failures\n--\n-- @since 1.3.1\ninfoFailFormat :: ConsoleFormat\ninfoFailFormat = ConsoleFormat NormalIntensity Dull  Red\n\n-- | Format used to display sucesses\n--\n-- @since 1.3.1\nokFormat :: ConsoleFormat\nokFormat = ConsoleFormat NormalIntensity Dull  Green\n\n-- | Format used to display additional information on sucesses\n--\n-- @since 1.3.1\ninfoOkFormat :: ConsoleFormat\ninfoOkFormat = ConsoleFormat NormalIntensity Dull  White\n\n-- | Format used to display skipped tests\n--\n-- @since 1.3.1\nskippedFormat :: ConsoleFormat\nskippedFormat = ConsoleFormat NormalIntensity Dull  Magenta\n'
