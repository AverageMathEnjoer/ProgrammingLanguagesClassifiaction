b'\xef\xbb\xbfnamespace RonSijm.ButtFish.Encoders;\n\n/// <summary>\n/// Config for TimeUnits.\n/// *Morse Code timing rules*\n/// There are rules to help people distinguish dots from dashes in Morse code.\n/// - The length of a dot is 1 time unit.\n/// - A dash is 3 time units.\n/// - The space between symbols(dots and dashes) of the same letter is 1 time unit.\n/// - The space between letters is 3 time units.\n/// - The space between words is 7 time units.\n/// Source: https://www.codebug.org.uk/learn/step/541/morse-code-timing-rules/\n/// </summary>\npublic static class TimeUnitConfig\n{\n    public static int TimeInit { get; set; }\n\n    public static int DotTime => TimeInit;\n    public static int DashTime => TimeInit * 3;\n    public static int SpaceBetweenSymbols => TimeInit;\n    public static int SpaceBetweenLetters => TimeInit * 3;\n    public static int SpaceBetweenWords => TimeInit * 7;\n}'
