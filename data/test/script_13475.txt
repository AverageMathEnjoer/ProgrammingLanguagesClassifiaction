b'package effectivejava.chapter7.item48;\n\nimport java.math.BigInteger;\nimport java.util.stream.LongStream;\n\npublic class ParallelPrimeCounting {\n    // Prime-counting stream pipeline - parallel version (Page 225)\n    static long pi(long n) {\n        return LongStream.rangeClosed(2, n)\n                .parallel()\n                .mapToObj(BigInteger::valueOf)\n                .filter(i -> i.isProbablePrime(50))\n                .count();\n    }\n\n    public static void main(String[] args) {\n        System.out.println(pi(10_000_000));\n    }\n}\n'
