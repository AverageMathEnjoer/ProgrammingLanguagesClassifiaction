b'// UVa10214 Trees in a Wood\n// Rujia Liu\n// This problem asks for K/N, where N is the total number of trees in area |x|<=a, |y|<=b\n// The answer converges to 6/pi^2, see the problem description\n#include<cstdio>\n#include<cmath>\nusing namespace std;\n\nint phi(int n) {\n  int m = (int)sqrt(n+0.5);\n  int ans = n;\n  for(int i = 2; i <= m; i++) if(n % i == 0) {\n    ans = ans / i * (i-1);\n    while(n % i == 0) n /= i;\n  }\n  if(n > 1) ans = ans / n * (n-1);\n  return ans;\n}\n\nint gcd(int a, int b) {\n  return b == 0 ? a : gcd(b, a % b);\n}\n\nlong long f(int a, int b) {\n  long long ans = 0; // only for 1<=x<=a, 1<=y<=b\n  for(int x = 1; x <= a; x++) {\n    int k = b/x;\n    ans += phi(x) * k;\n    for(int y = k*x+1; y <= b; y++)\n      if(gcd(x, y) == 1) ans++;\n  }\n  return ans * 4 + 4;\n}\n\nint main() {\n  int a, b;\n  while(scanf("%d%d", &a, &b) == 2 && a) {\n    long long K = f(a,b);\n    long long N = (long long)(2*a+1) * (2*b+1) - 1;\n    printf("%.7lf\\n", (double)K / N);\n  }\n  return 0;\n}\n'
