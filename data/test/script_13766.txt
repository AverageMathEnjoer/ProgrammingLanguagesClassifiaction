b'package effectivejava.chapter6.item36;\n\nimport java.util.*;\n\n// EnumSet - a modern replacement for bit fields (Page 170)\npublic class Text {\n    public enum Style {BOLD, ITALIC, UNDERLINE, STRIKETHROUGH}\n\n    // Any Set could be passed in, but EnumSet is clearly best\n    public void applyStyles(Set<Style> styles) {\n        System.out.printf("Applying styles %s to text%n",\n                Objects.requireNonNull(styles));\n    }\n\n    // Sample use\n    public static void main(String[] args) {\n        Text text = new Text();\n        text.applyStyles(EnumSet.of(Style.BOLD, Style.ITALIC));\n    }\n}'
