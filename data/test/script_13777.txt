b'//https://www.youtube.com/watch?v=hZkb_Dqu7YY\nclass Solution {\npublic:\n    int jump(vector<int>& nums) {\n        int n = nums.size();\n        if(n < 2) return 0;\n        int maxIndexReachable = nums[0];\n        int ans = 1, lim = nums[0]; \n        \n        for(int i = 1; i < n; i++) {\n            if(i > lim) {\n                ans++;\n                lim = maxIndexReachable;\n            }\n            maxIndexReachable = max(maxIndexReachable, i+nums[i]);\n        }\n        return ans;\n    }\n};\n'
