b'{-# LANGUAGE DataKinds #-}\n{-# LANGUAGE KindSignatures #-}\n{-# LANGUAGE TypeOperators #-}\n\n-- | Servant specific additions that could be upstreamed\nmodule Cachix.Types.Servant\n  ( Get302,\n    Post302,\n    Head,\n  )\nwhere\n\nimport Protolude\nimport Servant.API\n\n-- Location header as per https://github.com/haskell-servant/servant/issues/117#issuecomment-381398666\ntype Get302 (cts :: [Type]) (hs :: [Type]) = Verb \'GET 302 cts (Headers (Header "Location" Text \': hs) NoContent)\n\ntype Post302 (cts :: [Type]) (hs :: [Type]) = Verb \'POST 302 cts (Headers (Header "Location" Text \': hs) NoContent)\n\n-- TODO: allow empty CT with HEAD\ntype Head = Verb \'HEAD 200 \'[JSON] NoContent\n'
