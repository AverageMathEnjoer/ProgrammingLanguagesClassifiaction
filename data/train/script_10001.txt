exiting ...\\n");\n    return 1;\n  }\n  barometer.loadCalibration();\n  float temperature;\n  float pressure;\n  float altitude;\n  while (true) {\n    barometer.setControl(BMP085_MODE_TEMPERATURE);\n    bcm2835_delay(5); // wait 5 ms for conversion \n    temperature = barometer.getTemperatureC();\n    barometer.setControl(BMP085_MODE_PRESSURE_3) ; //taking reading in highest accuracy measurement mode\n    bcm2835_delay( barometer.getMeasureDelayMicroseconds() / 1000 );\n    pressure = barometer.getPressure();\n    altitude = barometer.getAltitude(pressure);\n    printf("  Temperature: %3.1f deg C      Pressure %3.3f kPa      altitude %3.1f m\\r", temperature, pressure/1000.0, altitude);\n    fflush(stdout);\n    bcm2835_delay(200);\n  }\n  return 1; \n}\n'
