b'/*\n * @lc app=leetcode id=1 lang=cpp\n *\n * [1] Two Sum\n */\n\n// @lc code=start\n#include<bits/stdc++.h>\nclass Solution {\npublic:\n    vector<int> twoSum(vector<int>& nums, int target) {\n        map<int, int> id;\n        for(int i = 0; i < nums.size(); i++) {\n            int make = target - nums[i];\n            if(id[make]) {\n                return {id[make]-1, i};\n            }\n            id[nums[i]] = i+1;\n        }\n        return {-1, -1};\n    }\n};\n// @lc code=end\n\n'
