b'{-# LANGUAGE GeneralizedNewtypeDeriving #-}\n\n-- | Module contains definitions of pdf objects\n--\n-- See PDF1.7:7.3\n\nmodule Pdf.Core.Object\n( Object(..)\n, Dict\n, Array\n, Stream(..)\n, Ref(..)\n, Name\n)\nwhere\n\nimport Pdf.Core.Name (Name)\n\nimport Data.Int\nimport Data.ByteString (ByteString)\nimport Data.Scientific (Scientific)\nimport Data.Vector (Vector)\nimport Data.Hashable\nimport Data.HashMap.Strict as HashMap\n\n-- | Dictionary\ntype Dict = HashMap Name Object\n\n-- | An array\ntype Array = Vector Object\n\n-- | Contains stream dictionary and an offset in file\ndata Stream = S Dict Int64\n  deriving (Eq, Show)\n\n-- | Object reference, contains object index and generation\ndata Ref = R Int Int\n  deriving (Eq, Show, Ord)\n\ninstance Hashable Ref where\n  hashWithSalt salt (R a b) = hashWithSalt salt (a, b)\n\n-- | Any pdf object\ndata Object =\n  Number Scientific |\n  Bool Bool |\n  Name Name |\n  Dict Dict |\n  Array Array |\n  String ByteString |\n  Stream Stream |\n  Ref Ref |\n  Null\n  deriving (Eq, Show)\n'
