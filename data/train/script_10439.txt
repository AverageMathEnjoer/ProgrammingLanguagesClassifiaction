b'#\' print.inspectEM\n#\' \n#\' Print information from the EM algorithm to console.\n#\'\n#\' @usage \\method{print}{inspectEM}(x, ...)\n#\' @param x An \\code{inspectEM} object\n#\' @param ... Further arguments to be passed to \\code{print.fastLink()}.\n#\'\n#\' @export\n#\' @method print inspectEM\nprint.inspectEM <- function(x, ...){\n\n    ## ------\n    ## Output\n    ## ------\n    ## Details of match\n    cat("\\nMatched", x$nobs.a, "observations in dataset A to", x$nobs.b, "observations in dataset B.\\n")\n    cat("EM algorithm converged in", x$iter.converge, "iterations.\\n")\n    \n    ## Number of matches\n    min <- min(x$posterior.range)\n    max <- max(x$posterior.range)\n    cat(paste0("\\nNumber of matches found for posterior between ", min, " and ", max, ":\\n"))\n    cat(x$num.matches, "\\n")\n\n    ## Quality of each variable\n    cat("\\nProbability of observing pattern conditional on being in matched set\\n(By Variable):\\n")\n    print(x$matchprob.by.variable)\n    cat("\\n")\n\n    cat("\\nProbability of a match across all pairwise comparisons:\\n")\n    print(x$lambda)\n    cat("\\n")\n    \n    ## Posteriors\n    cat("\\nPosterior probability of a match, by matching pattern:\\n")\n    print(x$match.patterns)\n    cat("\\n")\n    \n}\n'
