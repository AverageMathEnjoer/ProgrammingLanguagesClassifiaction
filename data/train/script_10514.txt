b"class Solution {\npublic:\n    vector<int> minOperations(string boxes) {\n        int n = boxes.size();\n        int left_cnt = 0, tot_cnt = 0; // cnt of 1s\n        int left_sum = 0, tot_sum = 0; // sum of indices where its a 1\n        \n        \n        for(int i = 0; i < n; i++) {\n            if (boxes[i] == '1') tot_cnt++, tot_sum += i;    \n        }\n        \n        vector<int> ans = {};\n        for(int i = 0; i < n; i++) {\n            int right_cnt = tot_cnt - left_cnt;\n            int right_sum = tot_sum - left_sum;\n            \n            int left = left_cnt * i - left_sum;\n            int right = right_sum - right_cnt * i;\n            ans.push_back(left + right);\n            \n            if (boxes[i] == '1') left_cnt++, left_sum += i;\n        }  \n        return ans;\n    }\n};\n\n\n/*\ni=13\nj<13\ni-j1 + i-j2 => 2*i - (j1+j2)\n\nj>13\nj1-i + j2-i => (j1+j2) - 2 *(i)\n*/\n\n"
