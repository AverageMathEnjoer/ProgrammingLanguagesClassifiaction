b'// UVa12627 Erratic Expansion\n// Rujia Liu\n#include<iostream>\nusing namespace std;\n\n// how many red balloons after k hours\nlong long c(int i) {\n  return i == 0 ? 1 : c(i-1)*3;\n}\n\n// how many red balloons in the first i rows, after k hours\nlong long f(int k, int i) {\n  if(i == 0) return 0;\n  if(k == 0) return 1;\n\n  int k2 = 1 << (k-1);\n  if(i >= k2) return f(k-1, i-k2) + c(k-1)*2;\n  else return f(k-1, i) * 2;\n}\n\n// how many red balloons in the last i rows, after k hours\nlong long g(int k, int i) {\n  if(i == 0) return 0;\n  if(k == 0) return 1;\n\n  int k2 = 1 << (k-1);\n  if(i >= k2) return g(k-1, i-k2) + c(k-1);\n  else return g(k-1,i);\n}\n\nint main() {\n  int T, k, a, b;\n  cin >> T;\n  for(int kase = 1; kase <= T; kase++) {\n    cin >> k >> a >> b;\n    cout << "Case " << kase << ": " << f(k, b) - f(k, a-1) << "\\n";\n  }\n  return 0;\n}\n'
