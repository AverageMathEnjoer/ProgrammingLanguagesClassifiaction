b'// UVa10976 Fractions Again?!\n// Rujia Liu\n#include<cstdio>\n#include<vector>\nusing namespace std;\n\nint main() {\n  int k;\n  while(scanf("%d", &k) == 1 && k) {\n    vector<int> X, Y;\n    for(int y = k+1; y <= k*2; y++) {\n      // 1/k = 1/x + 1/y => x = ky/(y-k)\n      if(k*y%(y-k) == 0)\n        { X.push_back(k*y/(y-k)); Y.push_back(y); }\n    }\n    printf("%d\\n", X.size());\n    for(int i = 0; i < X.size(); i++)\n      printf("1/%d = 1/%d + 1/%d\\n", k, X[i], Y[i]);\n  }\n  return 0;\n}\n'
