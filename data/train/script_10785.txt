b'class Solution {\npublic:\n    bool canJump(vector<int>& nums) {\n        int n = nums.size();\n        vector<bool> allowed(n, 0); // O(n) extra space\n        allowed[0] = 1;\n        for (int i = 0; i < n; i++) {\n            if (allowed[i]) {\n                for(int j = min(n-1, i+nums[i]); j >= i+1; j--) {\n                    if(allowed[j]) break;\n                    allowed[j] = 1;\n                }\n            }\n        }\n        return allowed[n-1];\n    }\n};\n\n/*\n_ _ _ 7 _ _ 1 1 1 1 1 _ \n      i x x x x x x x\n      \nif i is allowerd and 100\n      i+1,i+2,...,i+100 <-- allowed\n      \nwhether n-1 is allowed <-- true / false\n      \n   */   \n      \n     \n'
