b'// UVa10288 Coupons\n// Rujia Liu\n#include<iostream>\n#include<sstream>\nusing namespace std;\n\ntypedef long long LL;\n\nLL gcd(LL a, LL b) {\n  if(!b) return a;\n  return gcd(b, a%b);\n}\n\nLL lcm(LL a, LL b) {\n  return a / gcd(a, b) * b;\n}\n\nint LL_len(LL x) {\n  stringstream ss;\n  ss << x;\n  return ss.str().length();\n}\n\nvoid print_chars(int cnt, char ch) {\n  while(cnt--) cout << ch;\n}\n\nvoid output(LL a, LL b, LL c) {\n  if(b == 0) cout << a << "\\n";\n  else {\n    int L1 = LL_len(a);\n    print_chars(L1+1, \' \'); cout << b << "\\n";\n    cout << a << " "; print_chars(LL_len(c), \'-\'); cout << "\\n";\n    print_chars(L1+1, \' \'); cout << c << "\\n";\n  }\n}\n\nint main() {\n  int n;\n  while(cin >> n && n) {\n    if(n == 1) { output(1, 0, 0); continue; }\n    LL x = 1;\n    for(int i = 2; i <= n-1; i++)\n      x = lcm(x, i);\n\n    // b/c = 1/(n-1) + ... + 1/2\n    LL c = x, b = 0;\n    for(int i = 2; i <= n-1; i++)\n      b += x / i;\n    b *= n;\n    LL g = gcd(b, c);\n    b /= g; c /= g;\n\n    // ans = a + b/c\n    LL a = 1 + n + b / c;\n    b %= c;\n    output(a, b, c);\n  }\n  return 0;\n}\n'
