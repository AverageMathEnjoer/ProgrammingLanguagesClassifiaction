b'package com.db.williamchart.renderer.executor\n\nimport com.db.williamchart.data.DataPoint\nimport com.db.williamchart.data.Frame\n\nclass GetVerticalBarFrames {\n\n    operator fun invoke(\n        innerFrame: Frame,\n        zeroPositionY: Float,\n        spacingBetweenBars: Float,\n        data: List<DataPoint>\n    ): List<Frame> {\n        val halfBarWidth =\n            (innerFrame.right - innerFrame.left - (data.size + 1) * spacingBetweenBars) /\n                data.size / 2\n\n        return data.map {\n            Frame(\n                left = it.screenPositionX - halfBarWidth,\n                top = it.screenPositionY,\n                right = it.screenPositionX + halfBarWidth,\n                bottom = zeroPositionY\n            )\n        }\n    }\n}\n'
