b'package effectivejava.chapter9.item60;\n\nimport java.math.BigDecimal;\n\npublic class BigDecimalChange {\n    public static void main(String[] args) {\n        final BigDecimal TEN_CENTS = new BigDecimal(".10");\n\n        int itemsBought = 0;\n        BigDecimal funds = new BigDecimal("1.00");\n        for (BigDecimal price = TEN_CENTS;\n             funds.compareTo(price) >= 0;\n             price = price.add(TEN_CENTS)) {\n            funds = funds.subtract(price);\n            itemsBought++;\n        }\n        System.out.println(itemsBought + " items bought.");\n        System.out.println("Money left over: $" + funds);\n    }\n}\n'
