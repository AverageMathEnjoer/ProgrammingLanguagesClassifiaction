b'class Solution {\npublic:\n    vector<int> processQueries(vector<int>& queries, int m) {\n        unordered_map<int, int> pos, rev_pos; // pos[3] - 10 => 3 occurs at index 10\n        for(int i = 1; i <= m; i++) {\n            pos[i] = i-1, rev_pos[i-1] = i;\n        }\n        \n        vector<int> ans = {};\n        for(int q: queries) {\n            ans.push_back(pos[q]);\n            // increment the pos of all nos from 0 to pos[q]-1\n            for (int i = pos[q]-1 ; i >= 0; i--) {\n                int cur = rev_pos[i];\n                pos[cur] = i+1; // shift to right side\n                rev_pos[i+1] = cur;\n            }\n            pos[q] = 0;\n            rev_pos[0] = q;\n        }\n        return ans;\n    }\n};\n// 1,2,3,4,5\n// 3,1,2,4,5\n\n// _ _ _ _ _ _ _ _ _ _ _ \n          // ^ '
