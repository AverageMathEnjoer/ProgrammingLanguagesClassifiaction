b'// UVa1152 4 Values Whose Sum is Zero\n// Rujia Liu\n#include<cstdio>\n#include<algorithm>\nusing namespace std;\n\nconst int maxn = 4000 + 5;\nint n, c, A[maxn], B[maxn], C[maxn], D[maxn], sums[maxn*maxn];\n\nint main() {\n  int T;\n  scanf("%d", &T);\n  while(T--) {\n    scanf("%d", &n);\n    for(int i = 0; i < n; i++)\n      scanf("%d%d%d%d", &A[i], &B[i], &C[i], &D[i]);\n    c = 0;\n    for(int i = 0; i < n; i++)\n      for(int j = 0; j < n; j++)\n        sums[c++] = A[i] + B[j];\n    sort(sums, sums+c);\n    long long cnt = 0;\n    for(int i = 0; i < n; i++)\n      for(int j = 0; j < n; j++)\n        cnt += upper_bound(sums, sums+c, -C[i]-D[j]) - lower_bound(sums, sums+c, -C[i]-D[j]);\n    printf("%lld\\n", cnt);\n    if(T) printf("\\n");\n  }\n  return 0;\n}\n'
