b'package effectivejava.chapter5.item31;\n\nimport java.util.ArrayList;\nimport java.util.Collection;\nimport java.util.List;\nimport java.util.Random;\n\n// Wildcard type for parameter that serves as an T producer (page 141)\npublic class Chooser<T> {\n    private final List<T> choiceList;\n    private final Random rnd = new Random();\n\n    public Chooser(Collection<? extends T> choices) {\n        choiceList = new ArrayList<>(choices);\n    }\n\n    public T choose() {\n        return choiceList.get(rnd.nextInt(choiceList.size()));\n    }\n\n    public static void main(String[] args) {\n        List<Integer> intList = List.of(1, 2, 3, 4, 5, 6);\n        Chooser<Number> chooser = new Chooser<>(intList);\n        for (int i = 0; i < 10; i++) {\n            Number choice = chooser.choose();\n            System.out.println(choice);\n        }\n    }\n}\n'
