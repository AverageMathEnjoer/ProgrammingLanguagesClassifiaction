b'class Solution {\npublic:\n    int maxSatisfaction(vector<int>& arr) {\n        int n = arr.size();\n        sort(arr.begin(), arr.end());\n        int ans = 0;\n        for(int i = 0; i < n; i++) {\n            int cur = 0;\n            // considering the suffix from pos i\n            // [arr[i], arr[i+1], ... arr[n-1]]\n               // 1         2              \n            for(int j = i; j < n; j++) {\n                cur += (j-i+1) * arr[j];\n            }\n            ans = max(ans, cur);\n        }\n        return ans;\n    }\n};\n\n/*\na1 <= a2 <= a3 .... // we have proved\n(a1,a2,...,ak) <-- reordering of satisfaction array + you can remove elements\nmax(a1 + 2*a2 + 3*a3 + k*ak)\n    \n    -9   -8  -1   0   5\n    \n    how long a prefix would you remove to get the max score\n*/\n'
