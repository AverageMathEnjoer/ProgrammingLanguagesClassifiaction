b'// This script is provided as an example of how you might manually parse a CSV file in F#.\n\nopen System\nopen System.IO\nEnvironment.CurrentDirectory <- __SOURCE_DIRECTORY__\n\nlet file = @"..\\..\\data\\FootballResults.csv"\ntype Result =\n    { Date : DateTime\n      Home : string\n      Away : string\n      HomeGoals : int\n      AwayGoals : int }\n\nlet data =\n    file\n    |> File.ReadAllLines\n    |> Seq.skip 1\n    |> Seq.map(fun row ->\n        let fields = row.Split \',\'\n        { Date = DateTime.ParseExact(fields.[0], "MM/dd/yyyy", null)\n          Home = fields.[1]\n          Away = fields.[2]\n          HomeGoals = int fields.[3]\n          AwayGoals = int fields.[4] })\n\ndata\n|> Seq.filter(fun row -> row.HomeGoals > row.AwayGoals)\n|> Seq.countBy(fun row -> row.Home)\n|> Seq.sortByDescending snd\n|> Seq.take 3\n'
