b'using hashmap = unordered_map<int, int>;\n\nclass Solution {\npublic:\n    int findPairs(vector<int>& nums, int k) {\n        hashmap cnt;\n        for(int x: nums) cnt[x]++;\n        \n        int ans = 0;\n        for(auto p: cnt) { // iterating on unique numbers of the array \n            int x = p.first;\n            // check x+k exists in the array\n            if(cnt.find(x+k) == cnt.end()) {\n                continue;\n            }\n            ans += (k==0) ? cnt[x+k] >= 2 : cnt[x+k] >= 1;\n        }\n        \n        return ans;\n    }\n};\n'
