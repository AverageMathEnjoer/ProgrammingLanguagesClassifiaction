b'// UVa10934 Dropping water balloons\n// Rujia Liu\n#include<iostream>\n#include<cstring>\nusing namespace std;\n\nconst int maxk = 100;\nconst int maxa = 63;\n\nunsigned long long d[maxk+1][maxa+1];\n\nint main() {\n  memset(d, 0, sizeof(d));\n  for(int i = 1; i <= maxk; i++)\n    for(int j = 1; j <= maxa; j++)\n      d[i][j] = d[i-1][j-1] + 1 + d[i][j-1];\n\n  int k;\n  unsigned long long n;\n  while(cin >> k >> n && k) {\n    int ans = -1;\n    for(int i = 1; i <= maxa; i++)\n      if(d[k][i] >= n) { ans = i; break; }\n    if(ans < 0) cout << "More than " << maxa << " trials needed.\\n";\n    else cout << ans << "\\n";\n  }\n  return 0;\n}\n'
