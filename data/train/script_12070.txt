b'// UVa136 Ugly Numbers\n// Rujia Liu\n#include<iostream>\n#include<vector>\n#include<queue>\n#include<set>\nusing namespace std;\ntypedef long long LL;\nconst int coeff[3] = {2, 3, 5};\n\nint main() {\n  priority_queue<LL, vector<LL>, greater<LL> > pq;\n  set<LL> s;\n  pq.push(1);\n  s.insert(1);\n  for(int i = 1; ; i++) {\n    LL x = pq.top(); pq.pop();\n    if(i == 1500) {\n      cout << "The 1500\'th ugly number is " << x << ".\\n";\n      break;\n    }\n    for(int j = 0; j < 3; j++) {\n      LL x2 = x * coeff[j];\n      if(!s.count(x2)) { s.insert(x2); pq.push(x2); }\n    }\n  }\n  return 0;\n}\n'
