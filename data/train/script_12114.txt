b'# note that this functionality can now be achieved with multiple calls (one for each column)\n# to the mapIds() function in AnnotationDbi >= 1.30\nconvertIDs <- function( ids, from, to, db, ifMultiple=c("putNA", "useFirst")) {\n  stopifnot( inherits( db, "AnnotationDb" ) )\n  ifMultiple <- match.arg( ifMultiple )\n  suppressWarnings( selRes <- AnnotationDbi::select(\n    db, keys=ids, keytype=from, columns=c(from,to) ) )\n  if ( ifMultiple == "putNA" ) {\n    duplicatedIds <- selRes[ duplicated( selRes[,1] ), 1 ]\n    selRes <- selRes[ ! selRes[,1] %in% duplicatedIds, ]\n  }\n  return( selRes[ match( ids, selRes[,1] ), 2 ] )\n}\n'
