b'context("zero_zero")\t\ntest_that("contrast of two groups with all zeros - LFC zeroed out", {\t\n\n  # test comparison of two groups with all zeros\n  dds <- makeExampleDESeqDataSet(m=8, n=100, sizeFactors=c(1,1,.5,.5,1,1,2,2))\n  dds$condition <- factor(rep(c("A","B","C","D"),each=2))\n  counts(dds)[1,] <- c(100L,110L,0L,0L,100L,110L,0L,0L)\n  counts(dds)[2,] <- rep(0L, 8)\n  #plot(counts(dds)[1,] ~ dds$condition)\n  dds <- DESeq(dds)\n\n  res <- results(dds, contrast=c("condition","D","B"))[1,]\n  expect_equal(res$log2FoldChange, 0)\n  res <- results(dds, contrast=c(0,-1,0,1))[1,]\n  expect_equal(res$log2FoldChange, 0)\n  # list works too, because it\'s converted to numeric\n  res <- results(dds, contrast=list("condition_D_vs_A","condition_B_vs_A"))[1,]\n  expect_equal(res$log2FoldChange, 0)\n  \n  res <- results(dds, name="condition_D_vs_A")[1,]\n  expect_true(res$log2FoldChange != 0)\n  res <- results(dds,c(0,0,0,1))[1,]\n  expect_true(res$log2FoldChange != 0)\n\n  # if all samples have 0, should be NA\n  res <- results(dds, contrast=c("condition","D","B"))[2,]\n  expect_true(is.na(res$log2FoldChange))\n  res <- results(dds, contrast=c(0,-1,0,1))[2,]\n  expect_true(is.na(res$log2FoldChange))\n\n})\n'
