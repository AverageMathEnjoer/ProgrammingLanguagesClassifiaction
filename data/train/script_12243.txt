b'package com.db.williamchart.renderer.executor\n\nimport com.db.williamchart.data.DataPoint\nimport com.db.williamchart.data.Frame\n\nclass GetHorizontalBarFrames {\n\n    operator fun invoke(\n        innerFrame: Frame,\n        zeroPositionX: Float,\n        spacingBetweenBars: Float,\n        data: List<DataPoint>\n    ): List<Frame> {\n        val halfBarWidth =\n            (innerFrame.bottom - innerFrame.top - (data.size + 1) * spacingBetweenBars) /\n                data.size / 2\n\n        return data.map {\n            Frame(\n                left = zeroPositionX,\n                top = it.screenPositionY - halfBarWidth,\n                right = it.screenPositionX,\n                bottom = it.screenPositionY + halfBarWidth\n            )\n        }\n    }\n}\n'
