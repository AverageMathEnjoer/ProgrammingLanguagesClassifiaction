b'package effectivejava.chapter8.item50;\nimport java.util.*;\n\n// Two attacks on the internals of an "immutable" period (232-3)\npublic class Attacks {\n    public static void main(String[] args) {\n        // Attack the internals of a Period instance  (Page 232)\n        Date start = new Date();\n        Date end = new Date();\n        Period p = new Period(start, end);\n        end.setYear(78);  // Modifies internals of p!\n        System.out.println(p);\n\n        // Second attack on the internals of a Period instance  (Page 233)\n        start = new Date();\n        end = new Date();\n        p = new Period(start, end);\n        p.end().setYear(78);  // Modifies internals of p!\n        System.out.println(p);\n    }\n}\n'
