Be sure to adjust any settings you need specifically\n * after initialization, especially the gain settings if you happen to be seeing\n * a lot of -4096 values (see the datasheet for mor information).\n */\nvoid HMC5883L::initialize() {\n    // write CONFIG_A register\n    I2Cdev::writeByte(devAddr, HMC5883L_RA_CONFIG_A,\n        (HMC5883L_AVERAGING_8 << (HMC5883L_CRA_AVERAGE_BIT - HMC5883L_CRA_AVERAGE_LENGTH + 1)) |\n        (HMC5883L_RATE_15     << (HMC5883L_CRA_RATE_BIT - HMC5883L_CRA_RATE_LENGTH + 1)) |\n        (HMC5883L_BIAS_NORMAL << (HMC5883L_CRA_BIAS_BIT - HMC5883L_CRA_BIAS_LENGTH + 1)));\n\n    // write CONFIG_B register\n    setGain(HMC5883L_GAIN_1090);\n    \n    // write MODE register\n    setMode(HMC5883L_MODE_SINGLE);\n}\n\n/** Verify the I2C connection.\n * Make sure the device is connected and responds as expected.\n * @return True if connection is valid, false otherwise\n */\nbool HMC5883L::testConnection() {\n    if (I2Cdev::readBytes(devAddr, HMC5883L_RA_ID_A, 3, buffer) == 3) {\n        return (buffer[0] == \'H\' && buffer[1] == \'4\' && buffer[2] == \'3\');\n    }\n    return false;\n}\n\n// CONFIG_A register\n\n/** Get number of samples averaged per measurement.\n * @return Current samples averaged per measurement (0-3 for 1/2/4/8 respectively)\n * @see HMC5883L_AVERAGING_8\n * @see HMC5883L_RA_CONFIG_A\n * @see HMC5883L_CRA_AVERAGE_BIT\n * @see HMC5883L_CRA_AVERAGE_LENGTH\n */\nuint8_t HMC5883L::getSampleAveraging() {\n    I2Cdev::readBits(devAddr, HMC5883L_RA_CONFIG_A, HMC5883L_CRA_AVERAGE_BIT, HMC5883L_CRA_AVERAGE_LENGTH, buffer);\n    return buffer[0];\n}\n/** Set number of samples averaged per measurement.\n * @param averaging New samples averaged per measurement setting(0-3 for 1/2/4/8 respectively)\n * @see HMC5883L_RA_CONFIG_A\n * @see HMC5883L_CRA_AVERAGE_BIT\n * @see HMC5883L_CRA_AVERAGE_LENGTH\n */\nvoid HMC5883L::setSampleAveraging(uint8_t averaging) {\n    I2Cdev::writeBits(devAddr, HMC5883L_RA_CONFIG_A, HMC5883L_CRA_AVERAGE_BIT, HMC5883L_CRA_AVERAGE_LENGTH, averaging);\n}\n/** Get data output rate value.\n * The Table below shows all selectable output rates in continuous measurement\n * mode. All three channels shall be measured within a given output rate. Other\n * output rates with maximum rate of 160 Hz can be achieved by monitoring DRDY\n * interrupt pin in single measurement mode.\n *\n * Value | Typical Data Output Rate (Hz)\n * ------+------------------------------\n * 0     | 0.75\n * 1     | 1.5\n * 2     | 3\n * 3     | 7.5\n * 4     | 15 (Default)\n * 5     | 30\n * 6     | 75\n * 7     | Not used\n *\n * @return Current I2Cdev::readBits(devAddr, HMC5883L_RA_CONFIG_A, HMC5883L_CRA_RATE_BIT, HMC5883L_CRA_RATE_LENGTH, buffer);\n    return buffer[0];\n}\n/** Set data output rate value.\n * @param rate Rate of data output rate) {\n    I2Cdev::writeBits(devAddr, HMC5883L_RA_CONFIG_A, HMC5883L_CRA_RATE_BIT, HMC5883L_CRA_RATE_LENGTH, rate);\n}\n/** Get measurement bias value.\n * @return Current bias value (0-2 for normal/positive/negative respectively)\n * @see HMC5883L_BIAS_NORMAL\n * @see HMC5883L_RA_CONFIG_A\n * @see HMC5883L_CRA_BIAS_BIT\n * @see HMC5883L_CRA_BIAS_LENGTH\n */\nuint8_t HMC5883L::getMeasurementBias() {\n    I2Cdev::readBits(devAddr, HMC5883L_RA_CONFIG_A, HMC5883L_CRA_BIAS_BIT, HMC5883L_CRA_BIAS_LENGTH, buffer);\n    return buffer[0];\n}\n/** Set measurement bias value.\n * @param bias New bias value (0-2 for normal/positive/negative respectively)\n * @see HMC5883L_BIAS_NORMAL\n * @see HMC5883L_RA_CONFIG_A\n * @see HMC5883L_CRA_BIAS_BIT\n * @see HMC5883L_CRA_BIAS_LENGTH\n */\nvoid HMC5883L::setMeasurementBias(uint8_t bias) {\n    I2Cdev::writeBits(devAddr, HMC5883L_RA_CONFIG_A, HMC5883L_CRA_BIAS_BIT, HMC5883L_CRA_BIAS_LENGTH, bias);\n}\n\n// CONFIG_B register\n\n/** Get magnetic field gain value.\n * The table below shows nominal gain settings. Use the "Gain" column to convert\n * counts to Gauss. Choose a lower gain value (higher GN#) when total field\n * strength causes overflow in one of the data output registers (saturation).\n * The data output range for all settings is 0xF800-0x07FF (-2048 - 2047).\n *\n * Value | Field Range | Gain (LSB/Gauss)\n * ------+-------------+-----------------\n * 0     | +/- 0.88 Ga | 1370\n * 1     | +/- 1.3 Ga  | 1090 (Default)\n * 2     | +/- 1.9 Ga  | 820\n * 3     | +/- 2.5 Ga  | 660\n * 4     | +/- 4.0 Ga  | 440\n * 5     | +/- 4.7 Ga  | 390\n * 6     | +/- 5.6 Ga  | 330\n * 7     | +/- 8.1 Ga  | 230\n *\n * @return Current magnetic field gain value\n * @see HMC5883L_GAIN_1090\n * @see HMC5883L_RA_CONFIG_B\n * @see HMC5883L_CRB_GAIN_BIT\n * @see HMC5883L_CRB_GAIN_LENGTH\n */\nuint8_t HMC5883L::getGain() {\n    I2Cdev::readBits(devAddr, HMC5883L_RA_CONFIG_B, HMC5883L_CRB_GAIN_BIT, HMC5883L_CRB_GAIN_LENGTH, buffer);\n    return buffer[0];\n}\n/** Set magnetic field gain value.\n * @param gain New magnetic field gain value\n * @see getGain()\n * @see HMC5883L_RA_CONFIG_B\n * @see HMC5883L_CRB_GAIN_BIT\n * @see HMC5883L_CRB_GAIN_LENGTH\n */\nvoid HMC5883L::setGain(uint8_t gain) {\n    // use this method to guarantee that bits 4-0 are set to zero, which is a\n    // requirement specified in the datasheet; it\'s actually more efficient than\n    // using the I2Cdev.writeBits method\n    I2Cdev::writeByte(devAddr, HMC5883L_RA_CONFIG_B, gain << (HMC5883L_CRB_GAIN_BIT - HMC5883L_CRB_GAIN_LENGTH + 1));\n}\n\n// MODE register\n\n/** Get measurement mode.\n * In continuous-measurement mode, the device continuously performs measurements\n * and places the result in the data register. This register value will clear when\n * after the next valid measurement is made. Note that this method automatically\n * clears the appropriate bit in the MODE register if Single mode is active.\n * @param x 16-bit signed integer container for X-axis heading\n * @param y 16-bit signed integer container for Y-axis heading\n * @param z 16-bit signed integer container for Z-axis heading\n * @see HMC5883L_RA_DATAX_H\n */\nvoid HMC5883L::getHeading(int16_t *x, int16_t *y, int16_t *z) {\n    I2Cdev::readBytes(devAddr, HMC5883L_RA_DATAX_H, 6, buffer);\n    if (mode == HMC5883L_MODE_SINGLE) I2Cdev::writeByte(devAddr, HMC5883L_RA_MODE, HMC5883L_MODE_SINGLE << (HMC5883L_MODEREG_BIT - HMC5883L_MODEREG_LENGTH + 1));\n    *x = (((int16_t)buffer[0]) << 8) | buffer[1];\n    *y = (((int16_t)buffer[4]) << 8) | buffer[5];\n    *z = (((int16_t)buffer[2]) << 8) | buffer[3];\n}\n/** Get X-axis heading measurement.\n * @return 16-bit signed integer with X-axis heading\n * @see HMC5883L_RA_DATAX_H\n */\nint16_t HMC5883L::getHeadingX() {\n    // each axis read requires that ALL axis registers be read, even if only\n    // one is used; this was not done ineffiently in the code by accident\n    I2Cdev::readBytes(devAddr, HMC5883L_RA_DATAX_H, 6, buffer);\n    if (mode == HMC5883L_MODE_SINGLE) I2Cdev::writeByte(devAddr, HMC5883L_RA_MODE, HMC5883L_MODE_SINGLE << (HMC5883L_MODEREG_BIT - HMC5883L_MODEREG_LENGTH + 1));\n    return (((int16_t)buffer[4]) << 8) | buffer[5];\n}\n/** Get Z-axis heading measurement.\n * @return 16-bit signed integer with Z-axis heading\n * @see HMC5883L_RA_DATAZ_H\n */\nint16_t HMC5883L::getHeadingZ() {\n    // each axis read requires that ALL axis registers be read, even if only\n    // one is used; this was not done ineffiently in the code by accident\n    I2Cdev::readBytes(devAddr, HMC5883L_RA_DATAX_H, 6, buffer);\n    if (mode == HMC5883L_MODE_SINGLE) I2Cdev::writeByte(devAddr, HMC5883L_RA_MODE, HMC5883L_MODE_SINGLE << (HMC5883L_MODEREG_BIT - HMC5883L_MODEREG_LENGTH + 1));\n    return (((int16_t)buffer[2]) << 8) | buffer[3];\n}\n\n// STATUS register\n\n/** Get data output register lock status.\n * This bit is set when this some but not all for of the six data output\n * registers have been read. When this bit is set, the six data output registers\n * are locked and any new data will not be placed in these register until one of\n * three conditions are met: one, all six bytes have been read or the mode\n * changed, two, the mode is changed, or three, the measurement configuration is\n * changed.\n * @return Data output register lock status\n * @see HMC5883L_RA_STATUS\n * @see HMC5883L_STATUS_LOCK_BIT\n */\nbool HMC5883L::getLockStatus() {\n    I2Cdev::readBit(devAddr, HMC5883L_RA_STATUS, HMC5883L_STATUS_LOCK_BIT, buffer);\n    return buffer[0];\n}\n/** Get data ready status.\n * This bit is set when data is written to all six data registers, and cleared\n * when the device initiates a write to the data output registers and after one\n * or more of the data output registers are written to.
