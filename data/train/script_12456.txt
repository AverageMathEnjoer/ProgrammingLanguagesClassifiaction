b'class Solution {\npublic:\n    bool canVisitAllRooms(vector<vector<int>>& rooms) {\n        int n = rooms.size();\n        vector<bool> vis(n, 0);\n        \n        queue<int> q;\n        q.push(0);\n        vis[0] = 1;\n        while(!q.empty()) {\n            int cur = q.front();\n            q.pop();\n            for(int child: rooms[cur]) {\n                if (!vis[child]) {\n                    vis[child] = 1;\n                    q.push(child);\n                }\n            }\n        }\n        for(int i = 0; i < n; i++) {\n            if(!vis[i]) return 0;\n        }\n        return 1;\n    }\n};\n\n/*\nInput: [[1,3],[3,0,1],[2],[0]]\n0 --> 1\n|\n|\n3\n*/\n'
