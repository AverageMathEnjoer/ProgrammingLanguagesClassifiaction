b'// Check Youtube video for video explanation\nclass Solution {\npublic:\n    int firstMissingPositive(vector<int> A)\n    {\n        int n = A.size();\n        for(int i = 0; i < n; ++ i) {\n            while(A[i] >= 1 && A[i] <= n && A[i] != A[A[i]-1])\n                swap(A[i], A[A[i] - 1]);\n        }\n        \n        for(int i = 0; i < n; ++ i)\n            if(A[i] != i + 1)\n                return i + 1;\n        \n        return n + 1;\n    }\n};\n'
