b'package effectivejava.chapter5.item32;\n\nimport java.util.ArrayList;\nimport java.util.List;\n\n// List as a typesafe alternative to a generic varargs parameter (page 149)\npublic class FlattenWithList {\n    static <T> List<T> flatten(List<List<? extends T>> lists) {\n        List<T> result = new ArrayList<>();\n        for (List<? extends T> list : lists)\n            result.addAll(list);\n        return result;\n    }\n\n    public static void main(String[] args) {\n        List<Integer> flatList = flatten(List.of(\n                List.of(1, 2), List.of(3, 4, 5), List.of(6,7)));\n        System.out.println(flatList);\n    }\n}\n'
