b'// UVa11572 Unique snowflakes\n// Rujia Liu\n#include<cstdio>\n#include<set>\n#include<algorithm>\nusing namespace std;\n\nconst int maxn = 1000000 + 5;\nint A[maxn];\n\nint main() {\n  int T, n;\n  scanf("%d", &T);\n  while(T--) {\n    scanf("%d", &n);\n    for(int i = 0; i < n; i++) scanf("%d", &A[i]);\n\n    set<int> s;\n    int L = 0, R = 0, ans = 0;\n    while(R < n) {\n      while(R < n && !s.count(A[R])) s.insert(A[R++]);\n      ans = max(ans, R - L);\n      s.erase(A[L++]);\n    }\n    printf("%d\\n", ans);\n  }\n  return 0;\n}\n'
