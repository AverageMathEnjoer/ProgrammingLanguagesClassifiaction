b'open System.IO\nopen System\n\nlet nextId  =\n  DirectoryInfo(__SOURCE_DIRECTORY__).GetFiles("*.md", SearchOption.AllDirectories)\n  |> Seq.map (fun f -> f.Name.ToLowerInvariant().Split([|"fst-";"fs-"; "-"|], StringSplitOptions.RemoveEmptyEntries))\n  |> Seq.choose (function\n    | [||]\n    | [|_|] -> None\n    | items -> \n      items \n      |> Array.map (fun v -> Int32.TryParse(v)) \n      |> Array.filter fst \n      |> Array.map snd \n      |> Array.tryHead\n  )\n  |> Seq.max \n  |> ((+) 1)\n\nprintfn $"next RFC ID: %04i{nextId}"'
