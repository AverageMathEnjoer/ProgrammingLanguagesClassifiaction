b'package effectivejava.chapter8.item55;\n\nimport java.util.Optional;\n\n// Avoiding unnecessary use of Optional\'s isPresent method (Page 252)\npublic class ParentPid {\n    public static void main(String[] args) {\n        ProcessHandle ph = ProcessHandle.current();\n\n        // Inappropriate use of isPresent\n        Optional<ProcessHandle> parentProcess = ph.parent();\n        System.out.println("Parent PID: " + (parentProcess.isPresent() ?\n                String.valueOf(parentProcess.get().pid()) : "N/A"));\n\n        // Equivalent (and superior) code using orElse\n        System.out.println("Parent PID: " +\n            ph.parent().map(h -> String.valueOf(h.pid())).orElse("N/A"));\n    }\n}\n'
